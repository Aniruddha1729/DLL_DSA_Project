Pseudocode

function insertionSort(arr):
    n = arr.length
    for i from 1 to n-1:
        key = arr[i]
        j = i - 1
        
        // Shift elements greater than key
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j = j - 1
        
        // Insert key at correct position
        arr[j + 1] = key
    
    return arr
